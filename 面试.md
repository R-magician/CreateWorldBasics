### Vue 性能优化方法

```
1.路由懒加载
    vue异步组件技术
    es提案的import()===>推荐
        指定了相同的webpackChunkName，会合并打包成一个js文件，把组件按组分块
        component:() => import(/* webpackChunkName: 'ImportFuncDemo' */ '@/components/index')
    webpack提供的require.ensure()
2.keep-alive缓存页面
3.使用v-show复用DOM，v-for遍历避免同时使用v-if
4.长列表性能优化
    如果数据不是动态的，就不要做响应式
    如果是大数据长列表可以采用虚拟滚动，只渲染部分区域的内容
5.事件的销毁：
    Vue组件销毁时，会自动销毁他本身的全部指令及事件监听器
6.图片懒加载
    对于图片过多的页面，为了加载页面速度。很多时候我们需要将页面未出现在可视区域的图片
    先不做加载，等到滚动到可视区域在加载
7.第三方插件按需引入：一般UI第三方组件库可以按需引入，避免体积较大
8.无状态组件标记为函数组件
9.子组件分割：变量本地化
10.服务端渲染
```

```javascript
let numFormat = (number) =>{
```

### 选择排序(两层嵌套,第一次找最小的，第二次找第二小的)
```javascript
let selectionSort = (arr) => {
    let len = arr.length;
    let minIndex, temp;
    for (let i = 0; i < len - 1; i++) {
        minIndex = i;
        for (let j = i + 1; j < len; j++) {
            if (arr[j] < arr[minIndex]) {     //寻找最小的数
                minIndex = j;                 //将最小数的索引保存
            }
        }
        temp = arr[i];
        arr[i] = arr[minIndex];
        arr[minIndex] = temp;
    }
    return arr;
}
```

### 二分查找算法(折半查找)

```javascript
let fn = (arr, s, e, flag) => {
  if (arr.indexOf(flag) == -1) {
    return -1;
  }
  let index = Math.floor((s + e) / 2);
  if (arr[s] < flag && flag < arr[index]) {
    return fn(arr, s, index, flag);
  } else if (arr[index] < flag && flag < arr[e]) {
    return fn(arr, index, arr.length - 1, flag);
  } else {
    if (arr[s] == flag) {
      return s;
    } else if (arr[e] == flag) {
      return e;
    }
    return index;
  }
};
let arr = [1, 22, 33, 44, 55, 66, 77, 88, 99, 100];
```

### 防抖

事件发生后 n 秒内只会执行一次，如果再次触发，会重新计算时间(比如回城)

```javascript
function de(fn, time) {
  let timer = null;
  return function () {
    if (timer) {
      clearTimeout(timer);
      timer = setTimeout(fn, timer);
    } else {
      timer = setTimeout(fn, timer);
    }
  };
}
```

### 节流

高频事件 n 秒内只会执行一次

```javascript
function fd(fn, time) {
  let flag = true;
  return function () {
    if (!flag) {
      return (flag = false);
    }
    flag = false;
    setTimeout(() => {
      fn();
      flag = true;
    }, time);
  };
}
```
